	\advance\leftskip by 2pc
	\hrule width 0pt height 1.7\baselineskip\hrulefill
	\hrule width 0pt height 1.7\baselineskip\hrulefill
	\par
	\medskip
	\endgroup

	{\small\strut\hfil #1\hfil}\kern1pt}\kern1pt\vrule}\hrule}}
}

	{
		\boxit{#1}\hfil\boxit{#2}\hfil\boxit{#3}\hfil
		\boxit{#4}\hfil\boxit{#5}
	}
}
